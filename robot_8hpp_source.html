430  bool joinMotionUnsafe(
431  std::unique_lock<std::mutex> &lock,
432  const std::optional<std::chrono::duration<Rep, Period>> &timeout = std::nullopt) {
433  while (motion_generator_running_) {
434  if (timeout.has_value()) {
435  if (control_finished_condition_.wait_for(lock, timeout.value()) == std::cv_status::timeout) {
436  return false;
437  }
438  } else {
439  control_finished_condition_.wait(lock);
440  }
441  }
442  if (control_thread_.joinable())
443  control_thread_.join();
444  if (control_exception_ != nullptr) {
445  auto control_exception = control_exception_;
446  control_exception_ = nullptr;
447  std::rethrow_exception(control_exception);
448  }
449  return true;

452  template<typename ControlSignalType>
453  void moveInternal(
454  const std::shared_ptr<Motion<ControlSignalType>> &motion,
455  const std::function<void(const ControlFunc<ControlSignalType> &)> &control_func_executor,
456  bool async) {
457  {
458  std::unique_lock<std::mutex> lock(control_mutex_);
459  if (is_in_control_unsafe() && motion_generator_running_) {
460  if (!std::holds_alternative<MotionGenerator<ControlSignalType>>(motion_generator_)) {
461  throw InvalidMotionTypeException("The type of motion cannot change during runtime. Please ensure that the "
462  "previous motion finished before using a new type of motion.");
463  } else {
464  std::get<MotionGenerator<ControlSignalType>>(motion_generator_).updateMotion(motion);
465  }
466  } else {
467  joinMotionUnsafe(lock);
468 
469  motion_generator_.emplace<MotionGenerator<ControlSignalType>>(this, motion);
470  auto motion_generator = &std::get<MotionGenerator<ControlSignalType>>(motion_generator_);
471  motion_generator->registerUpdateCallback(
472  [this](const franka::RobotState &robot_state, franka::Duration duration, double time) {
473  std::lock_guard<std::mutex> lock(this->state_mutex_);
474  current_state_ = robot_state;
475  });
476  motion_generator_running_ = true;
477  control_thread_ = std::thread(
478  [this, control_func_executor, motion_generator]() {
479  try {
480  bool done = false;
481  while (!done) {
482  control_func_executor(
483  [motion_generator](const franka::RobotState &rs, franka::Duration d) {
484  return (*motion_generator)(rs, d);
485  });
486  std::unique_lock<std::mutex> lock(control_mutex_);
487 
488  // This code is just for the case that a new motion is set just after the old one terminates. If this
489  // happens, we need to continue with this motion, unless an exception occurs.
490  done = !motion_generator->has_new_motion();
491  if (motion_generator->has_new_motion()) {
492  motion_generator->resetTimeUnsafe();
493  } else {
494  done = true;
495  motion_generator_running_ = false;
496  control_finished_condition_.notify_all();
497  }
498  }
499  } catch (...) {
500  std::unique_lock<std::mutex> lock(control_mutex_);
501  control_exception_ = std::current_exception();
502  motion_generator_running_ = false;
503  control_finished_condition_.notify_all();
504  }
505  }
506  );
507  }
508  }
509  if (!async)
510  joinMotion();
511  }
