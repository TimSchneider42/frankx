235  template<class Rep = long, class Period = std::ratio<1>>
236  bool joinMotionUnsafe(
237  std::unique_lock<std::mutex> &lock,
238  const std::optional<std::chrono::duration<Rep, Period>> &timeout = std::nullopt) {
239  while (motion_generator_running_) {
240  if (timeout.has_value()) {
241  if (control_finished_condition_.wait_for(lock, timeout.value()) == std::cv_status::timeout) {
242  return false;
243  }
244  } else {
245  control_finished_condition_.wait(lock);
246  }
247  }
248  if (control_thread_.joinable())
249  control_thread_.join();
250  if (control_exception_ != nullptr) {
251  auto control_exception = control_exception_;
252  control_exception_ = nullptr;
253  std::rethrow_exception(control_exception);
254  }
255  return true;
256  }

258  template<typename ControlSignalType>
259  void moveInternal(
260  const std::shared_ptr<Motion<ControlSignalType>> &motion,
261  const std::function<void(const ControlFunc<ControlSignalType> &)> &control_func_executor,
262  bool async) {
263  {
264  std::unique_lock<std::mutex> lock(control_mutex_);
265  if (is_in_control_unsafe() && motion_generator_running_) {
266  if (!std::holds_alternative<MotionGenerator<ControlSignalType>>(motion_generator_)) {
267  throw InvalidMotionTypeException("The type of motion cannot change during runtime. Please ensure that the "
268  "previous motion finished before using a new type of motion.");
269  } else {
270  std::get<MotionGenerator<ControlSignalType>>(motion_generator_).updateMotion(motion);
271  }
272  } else {
273  joinMotionUnsafe(lock);
274 
275  motion_generator_.emplace<MotionGenerator<ControlSignalType>>(this, motion);
276  auto motion_generator = &std::get<MotionGenerator<ControlSignalType>>(motion_generator_);
277  motion_generator->registerUpdateCallback(
278  [this](const franka::RobotState &robot_state, franka::Duration duration, double time) {
279  std::lock_guard<std::mutex> lock(this->state_mutex_);
280  current_state_ = robot_state;
281  });
282  motion_generator_running_ = true;
283  control_thread_ = std::thread(
284  [this, control_func_executor, motion_generator]() {
285  try {
286  bool done = false;
287  while (!done) {
288  control_func_executor(
289  [motion_generator](const franka::RobotState &rs, franka::Duration d) {
290  return (*motion_generator)(rs, d);
291  });
292  std::unique_lock<std::mutex> lock(control_mutex_);
293 
294  // This code is just for the case that a new motion is set just after the old one terminates. If this
295  // happens, we need to continue with this motion, unless an exception occurs.
296  done = !motion_generator->has_new_motion();
297  if (motion_generator->has_new_motion()) {
298  motion_generator->resetTimeUnsafe();
299  } else {
300  done = true;
301  motion_generator_running_ = false;
302  control_finished_condition_.notify_all();
303  }
304  }
305  } catch (...) {
306  std::unique_lock<std::mutex> lock(control_mutex_);
307  control_exception_ = std::current_exception();
308  motion_generator_running_ = false;
309  control_finished_condition_.notify_all();
310  }
311  }
312  );
313  }
314  }
315  if (!async)
316  joinMotion();
317  }
